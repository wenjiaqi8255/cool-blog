import { describe, it, expect } from 'vitest';
import { render, screen, fireEvent } from '@testing-library/react';
import PortfolioModal from './PortfolioModal';

const mockArticle = {
  id: 1,
  title: 'Test Project',
  slug: 'test-project',
  date: new Date('2024-01-15'),
  tags: ['portfolio', 'react'],
  excerpt: 'Test excerpt',
  body: '# Test Content\n\nThis is a test project.'
};

describe('PortfolioModal', () => {
  it('renders null when isOpen is false', () => {
    const { container } = render(
      <PortfolioModal isOpen={false} onClose={() => {}} article={null} />
    );
    expect(container.firstChild).toBeNull();
  });

  it('renders null when article is null', () => {
    const { container } = render(
      <PortfolioModal isOpen={true} onClose={() => {}} article={null} />
    );
    expect(container.firstChild).toBeNull();
  });

  it('renders article title when open with article', () => {
    render(
      <PortfolioModal isOpen={true} onClose={() => {}} article={mockArticle} />
    );
    expect(screen.getByText('Test Project')).toBeDefined();
  });

  it('renders article date when open with article', () => {
    render(
      <PortfolioModal isOpen={true} onClose={() => {}} article={mockArticle} />
    );
    expect(screen.getByText(/2024/)).toBeDefined();
  });

  it('renders tags when article has tags', () => {
    render(
      <PortfolioModal isOpen={true} onClose={() => {}} article={mockArticle} />
    );
    expect(screen.getByText('portfolio')).toBeDefined();
  });

  it('calls onClose when close button clicked', () => {
    const onClose = vi.fn();
    render(
      <PortfolioModal isOpen={true} onClose={onClose} article={mockArticle} />
    );

    const closeButton = document.querySelector('.modal-close');
    if (closeButton) {
      fireEvent.click(closeButton);
      expect(onClose).toHaveBeenCalled();
    }
  });

  it('locks body scroll when open', () => {
    render(
      <PortfolioModal isOpen={true} onClose={() => {}} article={mockArticle} />
    );
    expect(document.body.style.overflow).toBe('hidden');
  });

  it('unlocks body scroll when closed', () => {
    const { unmount } = render(
      <PortfolioModal isOpen={true} onClose={() => {}} article={mockArticle} />
    );
    unmount();
    expect(document.body.style.overflow).toBe('');
  });
});
