#### C. Database Query Optimization

```typescript
// In articles.ts
export async function listPortfolioArticles(): Promise<Article[]> {
  // Only select needed fields
  const result = await db
    .select({
      id: articles.id,
      title: articles.title,
      slug: articles.slug,
      date: articles.date,
      tags: articles.tags,
      excerpt: articles.excerpt,
      image: articles.image,
      // Don't fetch full body for card display
    })
    .from(articles)
    .where(
      and(
        eq(articles.status, 'published'),
        isNull(articles.deleted_at),
        sql`${articles.tags} @> '{"Project"}'`
      )
    )
    .orderBy(desc(articles.date))
    .limit(12);

  return result;
}
```

#### D. Caching

```typescript
// Cache portfolio articles for 60 seconds
let cachedArticles: Article[] | null = null;
let cacheExpiry = 0;

export async function listPortfolioArticlesCached(): Promise<Article[]> {
  const now = Date.now();

  if (cachedArticles && now < cacheExpiry) {
    return cachedArticles;
  }

  cachedArticles = await listPortfolioArticles();
  cacheExpiry = now + 60000; // 60 seconds

  return cachedArticles;
}
```
