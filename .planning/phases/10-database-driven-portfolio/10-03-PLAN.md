# Plan 10-03: BentoGrid Integration

**Goal:** Replace static mockup cards with database-driven portfolio articles

**Duration:** 1.5 hours

**Dependencies:** Plan 10-02 (configuration system)

---

## Context

Current BentoGrid uses static cards from `src/config/cards.ts`. We need to integrate database queries and make it fully dynamic while preserving the visual aesthetic.

## Requirements

### Functional
- Load portfolio articles from database
- Apply configuration rules (featured, sizing, variants)
- Maintain responsive grid layout
- Preserve existing Bento aesthetic

### Technical
- Server-side rendering (SSR) in Astro
- Use existing BentoGrid component
- Map database articles to CardConfig format
- Preserve mobile responsiveness

## Implementation

### Step 1: Update Homepage

File: `src/pages/index.astro`

```typescript
---
import { pages } from '../config/content';
import BaseLayout from '../layouts/BaseLayout.astro';
import Header from '../components/layout/Header.astro';
import BentoGrid from '../components/layout/BentoGrid.astro';
import Footer from '../components/layout/Footer.astro';
import PortfolioModal from '../components/interactive/PortfolioModal';
import { listPortfolioArticles, type Article } from '../lib/articles';
import { mapArticlesToCards, portfolioConfig } from '../config/portfolio';
import type { CardConfig } from '../config/cards';

const currentPath = Astro.url.pathname;

// Fetch portfolio articles
const portfolioArticles: Article[] = await listPortfolioArticles();

// Map to card configurations
const cards: CardConfig[] = mapArticlesToCards(portfolioArticles, portfolioConfig);

// Determine if we should show animation (no articles)
const showAnimation = cards.length === 0;
---

<BaseLayout
  title={`${pages.home.title} | Home`}
  description={pages.home.description}
  type="website"
>
  <div class="layout-wrapper">
    <Header currentPath={currentPath} />

    <main class="bento-grid-container">
      {showAnimation ? (
        <div class="portfolio-animation">
          <!-- User will provide HTML animation -->
          <slot name="animation" />
        </div>
      ) : (
        <BentoGrid cards={cards} />
      )}
    </main>

    <Footer />
  </div>

  <!-- Portfolio Modal -->
  <PortfolioModal
    client:load
    articles={portfolioArticles.map((a) => ({
      ...a,
      date: a.date instanceof Date ? a.date.toISOString() : a.date
    }))}
  />

  <script>
    // Client-side interactivity for portfolio cards
    document.querySelectorAll('.bento-card').forEach((card) => {
      card.addEventListener('click', (e) => {
        const link = card.querySelector('a');
        if (link) {
          // If card has a link, let it navigate naturally
          return;
        }

        // Otherwise, check if it's a portfolio card
        const cardId = card.getAttribute('data-card-id');
        if (cardId) {
          window.dispatchEvent(new CustomEvent('open-portfolio-modal', {
            detail: { slug: cardId }
          }));
        }
      });
    });
  </script>
</BaseLayout>

<style>
  .layout-wrapper {
    max-width: 1600px;
    margin: 0 auto;
    padding: 40px;
  }

  .bento-grid-container {
    margin-top: 40px;
  }

  .portfolio-animation {
    min-height: 600px;
    display: flex;
    align-items: center;
    justify-content: center;
  }

  @media (max-width: 768px) {
    .layout-wrapper {
      padding: 16px;
    }

    .bento-grid-container {
      margin-top: 24px;
    }
  }
</style>
```

### Step 2: Update BentoGrid Component

File: `src/components/layout/BentoGrid.astro`

```typescript
---
import type { CardConfig } from '../../config/cards';
import ImageCard from '../cards/ImageCard.astro';
import TextCard from '../cards/TextCard.astro';
import TerminalCard from '../cards/TerminalCard.astro';
import StatsCard from '../cards/StatsCard.astro';

interface Props {
  cards: CardConfig[];
}

const { cards } = Astro.props;

// Map card types to components
const cardComponents = {
  image: ImageCard,
  text: TextCard,
  terminal: TerminalCard,
  stats: StatsCard
};
---

<div class="bento-grid">
  {cards.map((card) => {
    const Component = cardComponents[card.type];

    return (
      <div
        class:list={[
          'bento-card',
          `col-span-${card.span}`,
          card.row === 2 && 'row-span-2'
        ]}
        data-card-id={card.id}
        data-card-type={card.type}
      >
        <Component {...card.props} variant={card.variant} />
      </div>
    );
  })}
</div>

<style>
  /* ... existing styles remain unchanged ... */
</style>
```

### Step 4: Update Card Components

File: `src/components/cards/ImageCard.astro`

---
export interface Props {
  image?: string;
  title: string;
  metaTag?: string;
  link?: string;
}

const { image, title, metaTag, link } = Astro.props;

// If no image, render as text card instead
if (!image) {
  return (
    <div class="card text-card dark">
      {metaTag && <span class="meta-tag">{metaTag}</span>}
      <h3>{title}</h3>
      {link && <a href={link}>Read more →</a>}
    </div>
  );
}
---

<div class="card image-card">

  <img src={image} alt={title} loading="lazy" />
  <h3>{title}</h3>
  {link && <a href={link}>View Project →</a>}
</div>
