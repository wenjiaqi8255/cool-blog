## Context

Current PortfolioModal doesn't match the Bento grid aesthetic. Need to redesign it to use the same visual language (dark theme, borders, fonts, etc.).

## Requirements

### Functional
- Modal displays article content (title, date, tags, body)
- Code syntax highlighting works
- Markdown renders correctly
- Close on click outside or ESC key
- Keyboard accessible

### Visual
- Match Bento card aesthetic:
  - Dark background: `#1a1a2e`
  - Border: `#2d2d44`
  - Accent: `#00d4ff`
  - Monospace font for code
- Smooth animations
- Mobile responsive

### Technical
- React component (client-side)
- Uses existing Shiki highlighting
- Accessible (ARIA attributes)
- Focus trap when open
- **Security:** Sanitize HTML to prevent XSS

```typescript
import { useEffect, useRef, useState } from 'react';
import type { Article } from '../../lib/articles';
import DOMPurify from 'dompurify';

interface PortfolioModalProps {
  articles: Array<Article & { date: string }>;
}

export default function PortfolioModal({ articles }: PortfolioModalProps) {
  const [isOpen, setIsOpen] = useState(false);
  const [currentArticle, setCurrentArticle] = useState<Article | null>(null);
  const modalRef = useRef<HTMLDivElement>(null);
  const closeButtonRef = useRef<HTMLButtonElement>(null);

  // Listen for open event
  useEffect(() => {
    const handleOpen = (e: CustomEvent<{ slug: string }>) => {
      const article = articles.find((a) => a.slug === e.detail.slug);
      if (article) {
        setCurrentArticle(article);
        setIsOpen(true);
      }
    };

    window.addEventListener('open-portfolio-modal', handleOpen as EventListener);
    return () => {
      window.removeEventListener('open-portfolio-modal', handleOpen as EventListener);
    };
  }, [articles]);

  // Handle ESC key
  useEffect(() => {
    const handleEsc = (e: KeyboardEvent) => {
      if (e.key === 'Escape') {
        setIsOpen(false);
      }
    };

    if (isOpen) {
      document.addEventListener('keydown', handleEsc);
      document.body.style.overflow = 'hidden';

      // Focus close button
      setTimeout(() => closeButtonRef.current?.focus(), 100);
    }

    return () => {
      document.removeEventListener('keydown', handleEsc);
      document.body.style.overflow = '';
    };
  }, [isOpen]);

  // Handle click outside
  const handleBackdropClick = (e: React.MouseEvent) => {
    if (e.target === modalRef.current) {
      setIsOpen(false);
    }
  };

  if (!isOpen || !currentArticle) return null;

  // Render markdown safely
  const renderedBody = renderMarkdown(currentArticle.body);
  const sanitizedBody = DOMPurify.sanitize(renderedBody);

  return (
    <div
      ref={modalRef}
      className="modal-backdrop"
      onClick={handleBackdropClick}
      role="dialog"
      aria-modal="true"
      aria-labelledby="modal-title"
    >
      <div className="modal-content">
        {/* Header */}
        <div className="modal-header">
          <div className="meta">
            {currentArticle.tags
              ?.filter((t) => t !== 'Project' && t !== 'featured')
              .map((tag) => (
                <span className="tag" key={tag}>
                  {tag}
                </span>
              ))}
            <time className="date">
              {new Date(currentArticle.date).toLocaleDateString('en-US', {
                year: 'numeric',
                month: 'short',
                day: 'numeric'
              })}
            </time>
          </div>

          <button
            ref={closeButtonRef}
            className="close-button"
            onClick={() => setIsOpen(false)}
            aria-label="Close modal"
          >
            ×
          </button>
        </div>

        {/* Title */}
        <h2 id="modal-title" className="modal-title">
          {currentArticle.title}
        </h2>

        {/* Excerpt */}
        {currentArticle.excerpt && (
          <p className="modal-excerpt">{currentArticle.excerpt}</p>
        )}

        {/* Body - Rendered Markdown (SANITIZED) */}
        <div
          className="modal-body prose"
          dangerouslySetInnerHTML={{ __html: sanitizedBody }}
        />

        {/* Footer */}
        <div className="modal-footer">
          <a
            href={`/articles/${currentArticle.slug}`}
            className="read-more-link"
          >
            Read full article →
          </a>
        </div>
      </div>
    </div>
  );
}

// Markdown renderer with Shiki highlighting
function renderMarkdown(body: string): string {
  // TODO: Use markdown-it with Shiki for syntax highlighting
  // This will be implemented using existing setup from Phase 8
  return body;
}
```

### Step 4: Keyboard Navigation

```typescript
// In modal component
useEffect(() => {
  if (!isOpen) return;

  const handleTab = (e: KeyboardEvent) => {
    if (e.key !== 'Tab') return;

    // Trap focus within modal
    const focusableElements = modalRef.current?.querySelectorAll(
      'button, a[href], input, select, textarea, [tabindex]:not([tabindex="-1"])'
    );

    if (!focusableElements) return;

    const firstElement = focusableElements[0] as HTMLElement;
    const lastElement = focusableElements[focusableElements.length - 1] as HTMLElement;

    if (e.shiftKey && document.activeElement === firstElement) {
      e.preventDefault();
      lastElement.focus();
    } else if (!e.shiftKey && document.activeElement === lastElement) {
      e.preventDefault();
      firstElement.focus();
    }
  };

  document.addEventListener('keydown', handleTab);
  return () => document.removeEventListener('keydown', handleTab);
}, [isOpen]);
```

## Security Considerations

### XSS Prevention

**CRITICAL:** Never use `dangerouslySetInnerHTML` without sanitization!

✅ **CORRECT:**
```typescript
import DOMPurify from 'dompurify';

const sanitized = DOMPurify.sanitize(userContent);
<div dangerouslySetInnerHTML={{ __html: sanitized }} />
```

❌ **WRONG:**
```typescript
<div dangerouslySetInnerHTML={{ __html: userContent }} />
```

### Why DOMPurify?

- Removes malicious scripts
- Allows safe HTML tags (p, h1, code, etc.)
- Preserves syntax highlighting
- Industry-standard sanitizer
